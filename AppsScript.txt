        // Function to fetch weather data
        async function fetchWeatherData() {
            const temperatureElement = document.getElementById('temperature-cat');
            const weatherDescriptionElement = document.getElementById('weather-description-cat');
            const weatherIconElement = document.getElementById('weather-icon-cat');

            // ВАЖНО: Вам нужно заменить пустую строку на ваш действительный ключ API OpenWeatherMap.
            // Получите свой ключ API здесь: https://openweathermap.org/api
            const apiKey = '9966af9247ab4f97eb55cd6d29e201ce'; // <-- ВСТАВЬТЕ СЮДА ВАШ КЛЮЧ API OpenWeatherMap
            const city = 'Pattaya';
            const countryCode = 'TH';

            if (!apiKey) {
                console.error('OpenWeatherMap API Key is missing. Please insert your API key in the script.');
                temperatureElement.textContent = '--°C';
                weatherDescriptionElement.textContent = 'Нет ключа API';
                weatherIconElement.className = 'fas fa-exclamation-triangle mr-2';
                return; // Выходим из функции, если ключ API отсутствует
            }

            const apiUrl = `https://api.openweathermap.org/data/2.5/weather?q=${city},${countryCode}&units=metric&lang=ru&appid=${apiKey}`;

            try {
                const response = await fetch(apiUrl);
                if (!response.ok) {
                    // Проверяем конкретно ошибку 401 Unauthorized
                    if (response.status === 401) {
                        throw new Error('Unauthorized: Invalid OpenWeatherMap API Key. Please check your API key.');
                    }
                    throw new Error(`HTTP error! status: ${response.status}`);
                }
                const data = await response.json();

                const temperature = Math.round(data.main.temp);
                const description = data.weather[0].description;
                const iconCode = data.weather[0].icon;

                temperatureElement.textContent = `${temperature}°C`;
                weatherDescriptionElement.textContent = description.charAt(0).toUpperCase() + description.slice(1); // Делаем первую букву заглавной

                // Сопоставляем иконки OpenWeatherMap с иконками Font Awesome
                let faIconClass = 'fas fa-question-circle'; // Иконка по умолчанию для неизвестного
                if (iconCode.includes('01')) faIconClass = 'fas fa-sun'; // Ясное небо
                else if (iconCode.includes('02')) faIconClass = 'fas fa-cloud-sun'; // Мало облаков
                else if (iconCode.includes('03') || iconCode.includes('04')) faIconClass = 'fas fa-cloud'; // Рассеянные/разорванные облака
                else if (iconCode.includes('09') || iconCode.includes('10')) faIconClass = 'fas fa-cloud-showers-heavy'; // Дождь
                else if (iconCode.includes('11')) faIconClass = 'fas fa-bolt'; // Гроза
                else if (iconCode.includes('13')) faIconClass = 'fas fa-snowflake'; // Снег
                else if (iconCode.includes('50')) faIconClass = 'fas fa-smog'; // Туман

                weatherIconElement.className = faIconClass + ' mr-2'; // Применяем новый класс иконки
            } catch (error) {
                console.error('Error fetching weather data:', error);
                temperatureElement.textContent = '--°C';
                weatherDescriptionElement.textContent = 'Ошибка';
                weatherIconElement.className = 'fas fa-exclamation-triangle mr-2'; // Иконка ошибки
            }
        }

        // Загружаем данные о погоде при загрузке страницы
        document.addEventListener('DOMContentLoaded', fetchWeatherData);
    </script>

    <script>
        let quantities = {
            adult: 1,
            child: 0
        };
        let personalAssistantSelected = false;
        let selectedPayment = null;
        
        const prices = {
            adult: { baht: 900, rub: 2700, commission: 680 },
            child: { baht: 800, rub: 2400, commission: 680 },
            personalAssistant: 1000 // в рублях
        };
        
        function formatNumber(num) {
            return num.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ' ');
        }
        
        function formatCurrency(amount, currency) {
            const formatted = formatNumber(amount);
            return currency === 'rub' ? `${formatted}\u00A0₽` : `${formatted}${currency}`;
        }
        
        function changeQuantity(type, delta) {
            const newValue = quantities[type] + delta;
            if (newValue >= 0 && newValue <= 10) {
                quantities[type] = newValue;
                document.getElementById(type + '-qty').textContent = newValue;
                updatePrices();
            }
        }
        
        function togglePersonalAssistant() {
            personalAssistantSelected = !personalAssistantSelected;
            const checkbox = document.getElementById('assistant-checkbox');
            if (personalAssistantSelected) {
                checkbox.classList.add('checked');
            } else {
                checkbox.classList.remove('checked');
            }
            updatePrices();
        }
        
        function calculateTotal() {
            let totalBaht = quantities.adult * prices.adult.baht + quantities.child * prices.child.baht;
            return totalBaht;
        }
        
        function calculateCommission() {
            let commission = quantities.adult * prices.adult.commission + quantities.child * prices.child.commission;
            if (personalAssistantSelected) {
                commission += prices.personalAssistant;
            }
            return commission;
        }
        
        function calculateFullPrice() {
            let fullPrice = quantities.adult * prices.adult.rub + quantities.child * prices.child.rub;
            if (personalAssistantSelected) {
                fullPrice += prices.personalAssistant;
            }
            return fullPrice;
        }
        
        function calculateRemainingCost() {
            const remainingAdult = quantities.adult * 650;
            const remainingChild = quantities.child * 550;
            const remainingTotal = remainingAdult + remainingChild;
            return remainingTotal;
        }
        
        function calculateCommissionPercent() {
            const baseFullPrice = quantities.adult * prices.adult.rub + quantities.child * prices.child.rub;
            const baseCommission = quantities.adult * prices.adult.commission + quantities.child * prices.child.commission;
            return baseFullPrice > 0 ? Math.round((baseCommission / baseFullPrice) * 100) : 0;
        }
        
        function updatePrices() {
            const totalBaht = calculateTotal();
            const commissionRub = calculateCommission();
            const fullPriceRub = calculateFullPrice();
            const commissionPercent = calculateCommissionPercent();
            const remainingCost = calculateRemainingCost();
            
            document.getElementById('total-cost').textContent = `${totalBaht}฿`;
            document.getElementById('partial-price').textContent = formatCurrency(commissionRub, 'rub');
            document.getElementById('full-price').textContent = formatCurrency(fullPriceRub, 'rub');
            document.getElementById('commission-percent').textContent = `${commissionPercent}%`;
            document.getElementById('remaining-cost').textContent = `${remainingCost}฿`;
            
            // Обновляем информацию о рассрочке/долях
            const installmentInfo = document.getElementById('installment-info');
            if (fullPriceRub < 30000) {
                installmentInfo.innerHTML = `
                    Доступна оплата Долями 
                    <span class="installment-tooltip">
                        <div class="tooltip-text">
                            Разделите сумму покупки на <strong>4 платежа</strong>, сделав ваше путешествие еще доступнее (услуга предоставляется Т-Банком). 
                            <a href="https://ezyvizit.com/payment-rules" class="tooltip-link" target="_blank">Подробные условия</a>
                        </div>
                    </span>
                `;
            } else {
                installmentInfo.innerHTML = 'Доступна оплата в рассрочку';
            }
        }
        
        function selectPayment(type) {
            // Reset all radios
            document.getElementById('radio-partial').classList.remove('checked');
            document.getElementById('radio-full').classList.remove('checked');
            document.querySelectorAll('.payment-method').forEach(el => el.classList.remove('selected'));
            
            // Set selected
            selectedPayment = type;
            document.getElementById('radio-' + type).classList.add('checked');
            document.querySelector('.payment-method[onclick="selectPayment(\'' + type + '\')"]').classList.add('selected');
        }
        
        function bookNow() {
            const tooltip = document.getElementById('payment-tooltip');
            
            if (!selectedPayment) {
                tooltip.classList.add('show');
                setTimeout(() => {
                    tooltip.classList.remove('show');
                }, 3000);
                return;
            }
            
            const totalBaht = calculateTotal();
            const commissionRub = calculateCommission();
            const fullPriceRub = calculateFullPrice();
            
            if (selectedPayment === 'partial') {
                const remainingAdult = quantities.adult * 650;
                const remainingChild = quantities.child * 550;
                const remainingTotal = remainingAdult + remainingChild;
                alert(`Переход к оплате предоплаты: ${formatCurrency(commissionRub, 'rub')}\nОстальная сумма оплачивается гиду на месте: ${remainingTotal}฿`);
            } else {
                alert(`Переход к оплате полной стоимости: ${formatCurrency(fullPriceRub, 'rub')}`);
            }
        }
        
        // Initialize
        updatePrices();
    </script>
